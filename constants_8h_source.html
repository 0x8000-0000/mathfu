27 /// @file mathfu/constants.h
28 /// @brief Vector constants for specific dimensions.
29 /// @addtogroup mathfu_constants
30 ///
31 /// It is preferable to use these constants rather than constructing them
32 /// when they're required. Construction most-likely slower than loading them
33 /// from memory.
34 /// <p>
35 /// For example, the following:<br>
36 /// <code>
37 /// lookat = mat4::LookAt(target, position, mathfu::kAxisY3f);
38 /// </code>
39 /// <br>is preferable to:<br>
40 /// <code>
41 /// lookat = mat4::LookAt(target, position,
42 /// mathfu::Vector<float, 3>(0.0f, 1.0f, 0.0f));
43 /// </code>
44 /// <br> in terms of efficiency and in addition to resulting in more concise
45 /// code.
46 /// </p>
47 ///
48 /// Depending on your linker's sophistication and settings, these constants may
49 /// be duplicated in every compilation unit in which they're used. However,
50 /// most linkers should be able to detect and eliminate this duplication.
51 
52 /// @addtogroup mathfu_constants
53 /// @{
