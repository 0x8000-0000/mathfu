163  inline Vector<float, 3> operator*(const Vector<float, 3>& v) const {
164  Vector<float, 3> return_v;
165  Simd4fUnion temp_v;
166 #ifdef MATHFU_COMPILE_WITH_PADDING
167  temp_v.simd = v.data_.simd;
168  temp_v.float_array[3] = 1;
169  simd4x4f_matrix_vector_mul(&data_.simd_matrix, &temp_v.simd,
170  &return_v.data_.simd);
171  return_v *= (1 / return_v.data_.float_array[3]);
172 #else
173  temp_v.simd = simd4f_create(v[0], v[1], v[2], 1.0f);
174  simd4x4f_matrix_vector_mul(&data_.simd_matrix, &temp_v.simd, &temp_v.simd);
175  simd4f_mul(temp_v.simd, simd4f_splat(temp_v.float_array[3]));
176  MATHFU_VECTOR3_STORE3(temp_v.simd, return_v.data_);
177 #endif // MATHFU_COMPILE_WITH_PADDING
178  return return_v;
179  }

310  /// @brief Constructs a Matrix<float, 4> from an AffineTransform.
311  ///
312  /// @param affine An AffineTransform reference to be used to construct
313  /// a Matrix<float, 4> by adding in the 'w' row of [0, 0, 0, 1].
314  static inline Matrix<float, 4> FromAffineTransform(
315  const AffineTransform& affine) {
316  Matrix<float, 4> m;
317  m.data_.simd_matrix.x = simd4f_uload4(&affine[0]);
318  m.data_.simd_matrix.y = simd4f_uload4(&affine[4]);
319  m.data_.simd_matrix.z = simd4f_uload4(&affine[8]);
320  m.data_.simd_matrix.w = simd4f_uload4(&kAffineWColumn[0]);
321  return m.Transpose();
322  }

324  /// @brief Converts a Matrix<float, 4> into an AffineTransform.
325  ///
326  /// @param m A Matrix<float, 4> reference to be converted into an
327  /// AffineTransform by dropping the fixed 'w' row.
328  ///
329  /// @return Returns an AffineTransform that contains the essential
330  /// transformation data from the Matrix<float, 4>.
331  static inline AffineTransform ToAffineTransform(const Matrix<float, 4>& m) {
332  AffineTransform affine;
333  const Matrix<float, 4> mt = m.Transpose();
334  simd4f_ustore4(mt.data_.simd_matrix.x, &affine[0]);
335  simd4f_ustore4(mt.data_.simd_matrix.y, &affine[4]);
336  simd4f_ustore4(mt.data_.simd_matrix.z, &affine[8]);
337  return affine;
338  }
