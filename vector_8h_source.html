89 /// @class VectorPacked "mathfu/vector.h"
90 /// @brief Packed N-dimensional vector.
91 ///
92 /// Some Vector classes are padded so that it's possible to use the data
93 /// structures with SIMD instructions. This structure can be used in
94 /// conjunction with unpacked Vector classes to pack data
95 /// into flat arrays suitable for sending to a GPU (e.g vertex buffers).
96 ///
97 /// <p>
98 /// For example, to pack (store) an unpacked to packed vector:<br>
99 /// <blockquote><code><pre>
100 /// VectorPacked<float, 3> packed;
101 /// Vector<float, 3> vector(3, 2, 1);
102 /// vector.Pack(&packed);
103 /// </pre></code></blockquote>
104 /// or<br>
105 /// <blockquote><code><pre>
106 /// Vector<float, 3> vector(3, 2, 1);
107 /// VectorPacked<float, 3> packed = vector;
108 /// </pre></code></blockquote>
109 /// </p>
110 ///
111 /// <p>
112 /// To initialize a vector from a packed vector:<br>
113 /// <blockquote><code><pre>
114 /// VectorPacked<float, 3> packed = { 3, 2, 1 };
115 /// Vector<float, 3> vector(packed);
116 /// </pre></code></blockquote>
117 ///
118 /// @tparam T type of VectorPacked elements.
119 /// @tparam d dimensions (number of elements) in the VectorPacked structure.
120 template <class T, int d>
121 struct VectorPacked {
122  /// Create an uninitialized VectorPacked.
123  VectorPacked() {}
124 
125  /// Create a VectorPacked from a Vector.
126  ///
127  /// Both VectorPacked and Vector must have the same number of dimensions.
128  /// @param vector Vector to create the VectorPacked from.
129  explicit VectorPacked(const Vector<T, d>& vector) { vector.Pack(this); }
130 
131  /// Copy a Vector to a VectorPacked.
132  ///
133  /// Both VectorPacked and Vector must have the same number of dimensions.
134  /// @param vector Vector to copy to the VectorPacked.
135  /// @returns A reference to this VectorPacked.
136  VectorPacked& operator=(const Vector<T, d>& vector) {
137  vector.Pack(this);
138  return *this;
139  }
140 
141  /// Elements of the packed vector one per dimension.
142  T data[d];
143 };
144 /// @}

821 /// @cond MATHFU_INTERNAL
822 template <typename T, int d, typename CompatibleT>
823 static inline Vector<T, d> FromTypeHelper(const CompatibleT& compatible) {
824 // C++11 is required for constructed unions.
825 #if __cplusplus >= 201103L
826  // Use a union instead of reinterpret_cast to avoid aliasing bugs.
827  union ConversionUnion {
828  ConversionUnion() {} // C++11.
829  CompatibleT compatible;
830  VectorPacked<T, d> packed;
831  } u;
832  static_assert(sizeof(u.compatible) == d * sizeof(T), "Conversion size mismatch.");
833 
834  // The read of `compatible` and write to `u.compatible` gets optimized away,
835  // and this becomes essentially a safe reinterpret_cast.
836  u.compatible = compatible;
837 
838  // Call the packed vector constructor with the `compatible` data.
839  return Vector<T, d>(u.packed);
840 #else
841  // Use the less-desirable memcpy technique if C++11 is not available.
842  // Most compilers understand memcpy deep enough to avoid replace the function
843  // call with a series of load/stores, which should then get optimized away,
844  // however in the worst case the optimize away may not happen.
845  // Note: Memcpy avoids aliasing bugs because it operates via unsigned char*,
846  // which is allowed to alias any type.
847  // See:
848  // http://stackoverflow.com/questions/15745030/type-punning-with-void-without-breaking-the-strict-aliasing-rule-in-c99
849  Vector<T, d> v;
850  assert(sizeof(compatible) == d * sizeof(T));
851  memcpy(&v, &compatible, sizeof(compatible));
852  return v;
853 #endif // __cplusplus >= 201103L
854 }
855 /// @endcond

857 /// @cond MATHFU_INTERNAL
858 template <typename T, int d, typename CompatibleT>
859 static inline CompatibleT ToTypeHelper(const Vector<T, d>& v) {
860 // See FromTypeHelper() for comments.
861 #if __cplusplus >= 201103L
862  union ConversionUnion {

864  CompatibleT compatible;
865  VectorPacked<T, d> packed;
866  } u;
867  static_assert(sizeof(u.compatible) == d * sizeof(T), "Conversion size mismatch.");

870 #else

872  assert(sizeof(compatible) == d * sizeof(T));
873  memcpy(&compatible, &v, sizeof(compatible));
