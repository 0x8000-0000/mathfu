79 /// @class VectorPacked "mathfu/vector.h"
80 /// @brief Packed N-dimensional vector.
81 ///
82 /// Some Vector classes are padded so that it's possible to use the data
83 /// structures with SIMD instructions. This structure can be used in
84 /// conjunction with unpacked Vector classes to pack data
85 /// into flat arrays suitable for sending to a GPU (e.g vertex buffers).
86 ///
87 /// <p>
88 /// For example, to pack (store) an unpacked to packed vector:<br>
89 /// <blockquote><code><pre>
90 /// VectorPacked<float, 3> packed;
91 /// Vector<float, 3> vector(3, 2, 1);
92 /// vector.Pack(&packed);
93 /// </pre></code></blockquote>
94 /// or<br>
95 /// <blockquote><code><pre>
96 /// Vector<float, 3> vector(3, 2, 1);
97 /// VectorPacked<float, 3> packed = vector;
98 /// </pre></code></blockquote>
99 /// </p>
100 ///
101 /// <p>
102 /// To initialize a vector from a packed vector:<br>
103 /// <blockquote><code><pre>
104 /// VectorPacked<float, 3> packed = { 3, 2, 1 };
105 /// Vector<float, 3> vector(packed);
106 /// </pre></code></blockquote>
107 ///
108 /// @tparam T type of VectorPacked elements.
109 /// @tparam d dimensions (number of elements) in the VectorPacked structure.
110 template<class T, int d>
111 struct VectorPacked {
112  /// Create an uninitialized VectorPacked.
113  VectorPacked() {}
114 
115  /// Create a VectorPacked from a Vector.
116  ///
117  /// Both VectorPacked and Vector must have the same number of dimensions.
118  /// @param vector Vector to create the VectorPacked from.
119  explicit VectorPacked(const Vector<T, d> &vector) {
120  vector.Pack(this);
121  }
122 
123  /// Copy a Vector to a VectorPacked.
124  ///
125  /// Both VectorPacked and Vector must have the same number of dimensions.
126  /// @param vector Vector to copy to the VectorPacked.
127  /// @returns A reference to this VectorPacked.
128  VectorPacked& operator=(const Vector<T, d> &vector) {
129  vector.Pack(this);
130  return *this;
131  }
132 
133  /// Elements of the packed vector one per dimension.
134  T data[d];
135 };
