212  static inline Vector<float, 2> Lerp(const Vector<float, 2>& v1,
213  const Vector<float, 2>& v2,
214  float percent) {
215  const Vector<float, 2> percentv(percent);
216  const Vector<float, 2> one_minus_percent(
217  simd2f_sub(simd2f_splat(1.0f), percentv.data_.simd));
218  return Vector<float, 2>(
219  simd2f_add(simd2f_mul(one_minus_percent.data_.simd, v1.data_.simd),
220  simd2f_mul(percentv.data_.simd, v2.data_.simd)));
221  }
