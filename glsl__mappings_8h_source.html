82 /// @brief Calculate the cross product of two 3-dimensional Vectors.
83 ///
84 /// @param v1 Vector to multiply
85 /// @param v2 Vector to multiply
86 /// @return 3-dimensional vector that contains the result.
87 template<class T>
88 inline Vector<T, 3> cross(const Vector<T, 3>& v1, const Vector<T, 3>& v2) {
89  return Vector<T, 3>::CrossProduct(v1,v2);
90 }

96 /// @return Scalar dot product result.
97 template<class TV>
