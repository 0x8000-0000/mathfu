110  /// @brief Calculate the inverse Quaternion.
111  ///
112  /// This calculates the inverse such that <code>(q * q).Inverse()</code>
113  /// is the identity.
114  ///
115  /// @return Quaternion containing the result.
116  inline Quaternion<T> Inverse() const { return Quaternion<T>(s_, -v_); }

118  /// @brief Multiply this Quaternion with another Quaternion.
119  ///
120  /// @note This is equivalent to
121  /// <code>FromMatrix(ToMatrix() * q.ToMatrix()).</code>
122  /// @param q Quaternion to multiply with.
123  /// @return Quaternion containing the result.
124  inline Quaternion<T> operator*(const Quaternion<T>& q) const {
125  return Quaternion<T>(
126  s_ * q.s_ - Vector<T, 3>::DotProduct(v_, q.v_),
127  s_ * q.v_ + q.s_ * v_ + Vector<T, 3>::CrossProduct(v_, q.v_));
128  }

130  /// @brief Multiply this Quaternion by a scalar.
131  ///
132  /// This multiplies the angle of the rotation by a scalar factor.
133  /// @param s1 Scalar to multiply with.
134  /// @return Quaternion containing the result.
135  inline Quaternion<T> operator*(const T& s1) const {
136  T angle;
137  Vector<T, 3> axis;
138  ToAngleAxis(&angle, &axis);
139  angle *= s1;
140  return Quaternion<T>(cos(0.5f * angle),
141  axis.Normalized() * sin(0.5f * angle));
142  }
143 
144  /// @brief Multiply a Vector by this Quaternion.
145  ///
146  /// This will rotate the specified vector by the rotation specified by this
147  /// Quaternion.
148  ///
149  /// @param v1 Vector to multiply by this Quaternion.
150  /// @return Rotated Vector.
151  inline Vector<T, 3> operator*(const Vector<T, 3>& v1) const {
152  T ss = s_ + s_;
153  return ss * Vector<T, 3>::CrossProduct(v_, v1) + (ss * s_ - 1) * v1 +
154  2 * Vector<T, 3>::DotProduct(v_, v1) * v_;
155  }

157  /// @brief Normalize this quaterion (in-place).
158  ///
159  /// @return Length of the quaternion.
160  inline T Normalize() {
161  T length = sqrt(s_ * s_ + Vector<T, 3>::DotProduct(v_, v_));
162  T scale = (1 / length);
163  s_ *= scale;
164  v_ *= scale;
165  return length;
166  }

177  /// @brief Convert this Quaternion to an Angle and axis.
178  ///
179  /// The returned angle is the size of the rotation in radians about the
180  /// axis represented by this Quaternion.
181  ///
182  /// @param angle Receives the angle.
183  /// @param axis Receives the normalized axis.
184  inline void ToAngleAxis(T* angle, Vector<T, 3>* axis) const {
185  *axis = s_ > 0 ? v_ : -v_;
186  *angle = 2 * atan2(axis->Normalize(), s_ > 0 ? s_ : -s_);
187  }

189  /// @brief Convert this Quaternion to 3 Euler Angles.
190  ///
191  /// @return 3-dimensional Vector where each element is a angle of rotation
192  /// (in radians) around the x, y, and z axes.
193  inline Vector<T, 3> ToEulerAngles() const {
194  Matrix<T, 3> m(ToMatrix());
195  T cos2 = m[0] * m[0] + m[1] * m[1];
196  if (cos2 < 1e-6f) {
197  return Vector<T, 3>(0, m[2] < 0 ? static_cast<T>(0.5 * M_PI)
198  : static_cast<T>(-0.5 * M_PI),
199  -atan2(m[3], m[4]));
200  } else {
201  return Vector<T, 3>(atan2(m[5], m[8]), atan2(-m[2], sqrt(cos2)),
202  atan2(m[1], m[0]));
203  }
204  }

206  /// @brief Convert to a 3x3 Matrix.
207  ///
208  /// @return 3x3 rotation Matrix.
209  inline Matrix<T, 3> ToMatrix() const {
210  const T x2 = v_[0] * v_[0], y2 = v_[1] * v_[1], z2 = v_[2] * v_[2];
211  const T sx = s_ * v_[0], sy = s_ * v_[1], sz = s_ * v_[2];
212  const T xz = v_[0] * v_[2], yz = v_[1] * v_[2], xy = v_[0] * v_[1];
213  return Matrix<T, 3>(1 - 2 * (y2 + z2), 2 * (xy + sz), 2 * (xz - sy),
214  2 * (xy - sz), 1 - 2 * (x2 + z2), 2 * (sx + yz),
215  2 * (sy + xz), 2 * (yz - sx), 1 - 2 * (x2 + y2));
216  }

218  /// @brief Create a Quaternion from an angle and axis.
219  ///
220  /// @param angle Angle in radians to rotate by.
221  /// @param axis Axis in 3D space to rotate around.
222  /// @return Quaternion containing the result.
223  static Quaternion<T> FromAngleAxis(const T& angle, const Vector<T, 3>& axis) {
224  const T halfAngle = static_cast<T>(0.5) * angle;
225  Vector<T, 3> localAxis(axis);
226  return Quaternion<T>(cos(halfAngle),
227  localAxis.Normalized() * sin(halfAngle));
228  }

230  /// @brief Create a quaternion from 3 euler angles.
231  ///
232  /// @param angles 3-dimensional Vector where each element contains an
233  /// angle in radius to rotate by about the x, y and z axes.
234  /// @return Quaternion containing the result.
235  static Quaternion<T> FromEulerAngles(const Vector<T, 3>& angles) {
236  const Vector<T, 3> halfAngles(static_cast<T>(0.5) * angles[0],
237  static_cast<T>(0.5) * angles[1],
238  static_cast<T>(0.5) * angles[2]);
239  const T sinx = sin(halfAngles[0]), cosx = cos(halfAngles[0]);
240  const T siny = sin(halfAngles[1]), cosy = cos(halfAngles[1]);
241  const T sinz = sin(halfAngles[2]), cosz = cos(halfAngles[2]);
242  return Quaternion<T>(cosx * cosy * cosz + sinx * siny * sinz,
243  sinx * cosy * cosz - cosx * siny * sinz,
244  cosx * siny * cosz + sinx * cosy * sinz,
245  cosx * cosy * sinz - sinx * siny * cosz);
246  }

248  /// @brief Create a quaternion from a rotation Matrix.
249  ///
250  /// @param m 3x3 rotation Matrix.
251  /// @return Quaternion containing the result.
252  static Quaternion<T> FromMatrix(const Matrix<T, 3>& m) {
253  const T trace = m(0, 0) + m(1, 1) + m(2, 2);
254  if (trace > 0) {
255  const T s = sqrt(trace + 1) * 2;
256  const T oneOverS = 1 / s;
257  return Quaternion<T>(static_cast<T>(0.25) * s, (m[5] - m[7]) * oneOverS,
258  (m[6] - m[2]) * oneOverS, (m[1] - m[3]) * oneOverS);
259  } else if (m[0] > m[4] && m[0] > m[8]) {
260  const T s = sqrt(m[0] - m[4] - m[8] + 1) * 2;
261  const T oneOverS = 1 / s;
262  return Quaternion<T>((m[5] - m[7]) * oneOverS, static_cast<T>(0.25) * s,
263  (m[3] + m[1]) * oneOverS, (m[6] + m[2]) * oneOverS);
264  } else if (m[4] > m[8]) {
265  const T s = sqrt(m[4] - m[0] - m[8] + 1) * 2;
266  const T oneOverS = 1 / s;
267  return Quaternion<T>((m[6] - m[2]) * oneOverS, (m[3] + m[1]) * oneOverS,
268  static_cast<T>(0.25) * s, (m[5] + m[7]) * oneOverS);
269  } else {
270  const T s = sqrt(m[8] - m[0] - m[4] + 1) * 2;
271  const T oneOverS = 1 / s;
272  return Quaternion<T>((m[1] - m[3]) * oneOverS, (m[6] + m[2]) * oneOverS,
273  (m[5] + m[7]) * oneOverS, static_cast<T>(0.25) * s);
274  }
275  }

277  /// @brief Calculate the spherical linear interpolation between two
278  /// Quaternions.
279  ///
280  /// @param q1 Start Quaternion.
281  /// @param q2 End Quaternion.
282  /// @param s1 The scalar value determining how far from q1 and q2 the
283  /// resulting quaternion should be. A value of 0 corresponds to q1 and a
284  /// value of 1 corresponds to q2.
285  /// @result Quaternion containing the result.
286  static inline Quaternion<T> Slerp(const Quaternion<T>& q1,
287  const Quaternion<T>& q2, const T& s1) {
288  if (q1.s_ * q2.s_ + Vector<T, 3>::DotProduct(q1.v_, q2.v_) > 0.999999f)
289  return Quaternion<T>(q1.s_ * (1 - s1) + q2.s_ * s1,
290  q1.v_ * (1 - s1) + q2.v_ * s1);
291  return q1 * ((q1.Inverse() * q2) * s1);
292  }

312  /// @brief Returns a vector that is perpendicular to the supplied vector.
313  ///
314  /// @param v1 An arbitrary vector
315  /// @return A vector perpendicular to v1. Normally this will just be
316  /// the cross product of v1, v2. If they are parallel or opposite though,
317  /// the routine will attempt to pick a vector.
318  static inline Vector<T, 3> PerpendicularVector(const Vector<T, 3>& v) {
319  // We start out by taking the cross product of the vector and the x-axis to
320  // find something parallel to the input vectors. If that cross product
321  // turns out to be length 0 (i. e. the vectors already lie along the x axis)
322  // then we use the y-axis instead.
323  Vector<T, 3> axis = Vector<T, 3>::CrossProduct(
324  Vector<T, 3>(static_cast<T>(1), static_cast<T>(0), static_cast<T>(0)),
325  v);
326  // We use a fairly high epsilon here because we know that if this number
327  // is too small, the axis we'll get from a cross product with the y axis
328  // will be much better and more numerically stable.
329  if (axis.LengthSquared() < static_cast<T>(0.05)) {
330  axis = Vector<T, 3>::CrossProduct(
331  Vector<T, 3>(static_cast<T>(0), static_cast<T>(1), static_cast<T>(0)),
332  v);
333  }
334  return axis;
335  }

337  /// @brief Returns the a Quaternion that rotates from start to end.
338  ///
339  /// @param v1 The starting vector
340  /// @param v2 The vector to rotate to
341  /// @param preferred_axis the axis to use, if v1 and v2 are parallel.
342  /// @return A Quaternion describing the rotation from v1 to v2
343  /// See the comment on RotateFromToWithAxis for an explanation of the math.
344  static inline Quaternion<T> RotateFromToWithAxis(
345  const Vector<T, 3>& v1, const Vector<T, 3>& v2,
346  const Vector<T, 3>& preferred_axis) {
347  Vector<T, 3> start = v1.Normalized();
348  Vector<T, 3> end = v2.Normalized();
349 
350  T dot_product = Vector<T, 3>::DotProduct(start, end);
351  // Any rotation < 0.1 degrees is treated as no rotation
352  // in order to avoid division by zero errors.
353  // So we early-out in cases where it's less then 0.1 degrees.
354  //cos( 0.1 degrees) = 0.99999847691
355  if (dot_product >= static_cast<T>(0.99999847691)) {
356  return Quaternion<T>::identity;
357  }
358  // If the vectors point in opposite directions, return a 180 degree
359  // rotation, on the axis that they asked for.
360  if (dot_product <= static_cast<T>(-0.99999847691)) {
361  return Quaternion<T>(static_cast<T>(0), preferred_axis);
362  }
363  // Degenerate cases have been handled, so if we're here, we have to
364  // actually compute the angle we want:
365  Vector<T, 3> cross_product = Vector<T, 3>::CrossProduct(start, end);
366 
367  return Quaternion<T>(static_cast<T>(1.0) + dot_product, cross_product)
368  .Normalized();
369  }

371  /// @brief Returns the a Quaternion that rotates from start to end.
372  ///
373  /// @param v1 The starting vector
374  /// @param v2 The vector to rotate to
375  /// @return A Quaternion describing the rotation from v1 to v2. In the case
376  /// where the vectors are parallel, it returns the identity. In the case
377  /// where
378  /// they point in opposite directions, it picks an arbitrary axis. (Since
379  /// there
380  /// are technically infinite possible quaternions to represent a 180 degree
381  /// rotation.)
382  ///
383  /// The final equation used here is fairly elegant, but its derivation is
384  /// not obvious: We want to find the quaternion that represents the angle
385  /// between Start and End.
386  ///
387  /// The angle can be expressed as a quaternion with the values:
388  /// angle: ArcCos(dotproduct(start, end) / (|start|*|end|)
389  /// axis: crossproduct(start, end).normalized * sin(angle/2)
390  ///
391  /// or written as:
392  /// quaternion(cos(angle/2), axis * sin(angle/2))
393  ///
394  /// Using the trig identity:
395  /// sin(angle * 2) = 2 * sin(angle) * cos*angle)
396  /// Via substitution, we can turn this into:
397  /// sin(angle/2) = 0.5 * sin(angle)/cos(angle/2)
398  ///
399  /// Using this substitution, we get:
400  /// quaternion( cos(angle/2),
401  /// 0.5 * crossproduct(start, end).normalized
402  /// * sin(angle) / cos(angle/2))
403  ///
404  /// If we scale the whole thing up by 2 * cos(angle/2) then we get:
405  /// quaternion(2 * cos(angle/2) * cos(angle/2),
406  /// crossproduct(start, end).normalized * sin(angle))
407  ///
408  /// (Note that the quaternion is no longer normalized after this scaling)
409  ///
410  /// Another trig identity:
411  /// cos(angle/2) = sqrt((1 + cos(angle) / 2)
412  ///
413  /// Substituting this in, we can simplify the quaternion scalar:
414  ///
415  /// quaternion(1 + cos(angle),
416  /// crossproduct(start, end).normalized * sin(angle))
417  ///
418  /// Because cross(start, end) has a magnitude of |start|*|end|*sin(angle),
419  /// crossproduct(start,end).normalized
420  /// is equivalent to
421  /// crossproduct(start,end) / |start| * |end| * sin(angle)
422  /// So after that substitution:
423  ///
424  /// quaternion(1 + cos(angle),
425  /// crossproduct(start, end) / (|start| * |end|))
426  ///
427  /// dotproduct(start, end) has the value of |start| * |end| * cos(angle),
428  /// so by algebra,
429  /// cos(angle) = dotproduct(start, end) / (|start| * |end|)
430  /// we can replace our quaternion scalar here also:
431  ///
432  /// quaternion(1 + dotproduct(start, end) / (|start| * |end|),
433  /// crossproduct(start, end) / (|start| * |end|))
434  ///
435  /// If start and end are normalized, then |start| * |end| = 1, giving us a
436  /// final quaternion of:
437  ///
438  /// quaternion(1 + dotproduct(start, end), crossproduct(start, end))
439  static inline Quaternion<T> RotateFromTo(const Vector<T, 3>& v1,
440  const Vector<T, 3>& v2) {
441  Vector<T, 3> start = v1.Normalized();
442  Vector<T, 3> end = v2.Normalized();
443 
444  T dot_product = Vector<T, 3>::DotProduct(start, end);
445  // Any rotation < 0.1 degrees is treated as no rotation
446  // in order to avoid division by zero errors.
447  // So we early-out in cases where it's less then 0.1 degrees.
448  //cos( 0.1 degrees) = 0.99999847691
449  if (dot_product >= static_cast<T>(0.99999847691)) {
450  return Quaternion<T>::identity;
451  }
452  // If the vectors point in opposite directions, return a 180 degree
453  // rotation, on an arbitrary axis.
454  if (dot_product <= static_cast<T>(-0.99999847691)) {
455  return Quaternion<T>(0, PerpendicularVector(start));
456  }
457  // Degenerate cases have been handled, so if we're here, we have to
458  // actually compute the angle we want:
459  Vector<T, 3> cross_product = Vector<T, 3>::CrossProduct(start, end);
460 
461  return Quaternion<T>(static_cast<T>(1.0) + dot_product, cross_product)
462  .Normalized();
463  }
